for iName = 1:length(names)

    Name = names(iName);
    Path=zrp+Name+'/';

    try
        RGB=load(Path+Name+"_RGB.txt");
        RGB_nc=load(Path+Name+"_nc_RGB.txt");
        nc=0:dt:length(RGB)*dt-dt;
    catch
        disp("ERROR: Ошибка при формировании отчёта по "+Name);
        continue
    end
